Now, the number of series used for regression is changed. That is using data from last 2 months to last 16 months. Then the value of $R^2$ statistic is plotted against the number of months used in autoregression.

In [28]:
def autoReg(mon,data,name):
    X = data.shift(periods=30).reset_index().drop(range(0,mon*30))
    pasts = pd.DataFrame(data=X[name]).rename(columns={name: 'M1'})
    for i in range(1,mon):
        X = data.shift(periods=(i+1)*30).reset_index().drop(range(0,mon*30))
        pasts['M'+str(i+1)] = X[name] 
    
    y = data[name].reset_index().drop(range(0,mon*30)).drop(columns=['Date'])
    model = LinearRegression().fit(pasts, y)
    m = model.coef_[0]
    b = model.intercept_
    z = m[0]*pasts['M1']
    for i in range(1,mon):
        z += m[i]*pasts['M'+str(i+1)]
    
    return [y[name],z + b]

num_month = 16
R2s = []
for i in range(2,num_month+1):
    pas = autoReg(i,stck_norm,'HMC')
    r_squared = sklearn.metrics.r2_score(pas[0],pas[1],multioutput='uniform_average')

plt.plot(range(2,num_month+1),R2s,color='darkgreen',linewidth=1.1)
